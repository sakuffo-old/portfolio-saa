Collatz Sequence

This ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit. 


def collatz(number)
  i = 0
  until number == 1 do
    number = (number % 2 == 0) ?  number/2 : (3 * number) + 1
    i += 1
  end

  return i
end

def collatz_length(last_number)
  longest = 1
  number = 1

  1.upto(last_number) do |n|
    temp = collatz(n)
    if longest <= temp
      longest = temp
      number = n
    end
  end

  puts longest
  puts number
end

collatz_end_num = gets.chomp.to_i
